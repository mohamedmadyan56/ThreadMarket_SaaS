2. auth.controller.ts — استقبال الطلب

// modules/auth/auth.controller.ts
import { Request, Response } from "express";
import { authService } from "./auth.service";
import { StatusCodes } from "http-status-codes";
import { asyncHandler } from "../../shared/utils/asyncHandler";

export const refreshAccessToken = asyncHandler(
  async (req: Request, res: Response) => {
    // ===============================================
    // 1. الـ Controller هو اللي يقرا الـ cookies
    // ===============================================
    const incomingRefreshToken = req.cookies?.refreshToken;

    if (!incomingRefreshToken) {
      return res.status(401).json({
        success: false,
        message: "Refresh token is missing"
      });
    }

    // ===============================================
    // 2. استدعاء Service — ويرجع data عادي
    // ===============================================
    const result = await authService.refreshAccessToken(incomingRefreshToken);

    // ===============================================
    // 3. الـ Controller هو اللي يتعامل مع الـ cookies
    // ===============================================
    res.cookie("accessToken", result.accessToken, {
      httpOnly: true,
      secure: process.env.NODE_ENV === "production",
      sameSite: "lax",
      path: "/",
      maxAge: 24 * 60 * 60 * 1000, // 24 ساعة
    });

    res.cookie("refreshToken", result.refreshToken, {
      httpOnly: true,
      secure: process.env.NODE_ENV === "production",
      sameSite: "lax",
      path: "/",
      maxAge: 7 * 24 * 60 * 60 * 1000, // 7 أيام
    });

    // ===============================================
    // 4. الـ Controller يرد Response
    // ===============================================
    return res.status(200).json({
      success: true,
      data: {
        accessToken: result.accessToken,
        refreshToken: result.refreshToken,
      },
      message: "Token refreshed successfully",
    });
  }
);
✅ واضح؟ الـ Controller مسؤول عن:
1. قراءة data من الـ request (cookies, body, params)
2. استدعاء الـ Service
3. التعامل مع الـ Response (cookies, status, json)
مش مسؤول عن: Business Logic، التحقق من صحة التوكن، قاعدة البيانات.

3. auth.service.ts — Business Logic

// modules/auth/auth.service.ts
import { authModel } from "./auth.model";
import { tokenService } from "./token.service";
import { AppError } from "../../shared/errors/AppError";
import jwt from "jsonwebtoken";
import { ENV } from "../../config/env";

class AuthService {

  async refreshAccessToken(incomingRefreshToken: string) {
    // ===============================================
    // الخطوة 1: التحقق من صحة JWT signature
    // ===============================================
    // jwt.verify يتأكد من:
    //   - التوقيع سليم (matching REFRESH_TOKEN_SECRET)
    //   - التوكن لسه صالح (مخلصش)
    // لو أي حاجة غلط → يحصل exception
    let payload: any;
    try {
      payload = jwt.verify(incomingRefreshToken, ENV.REFRESH_TOKEN_SECRET!);
    } catch (error) {
      throw new AppError("Invalid or expired refresh token", 401);
    }

    // ===============================================
    // الخطوة 2: جلب المستخدم من قاعدة البيانات
    // ===============================================
    const user = await authModel.findById(payload.id);

    if (!user) {
      throw new AppError("User not found", 404);
    }

    // ===============================================
    // الخطوة 3: مقارنة التوكن مع المخزن في DB
    // ===============================================
    // ده أهم خطوة في الأمان: لو الـ refresh token اللي في DB
    ">مختلف عن اللي جاي من المستخدم → معناه إن التوكن اتلغى
    // (مثلاً بسبب rotation سابق أو logout)
    if (user.refreshToken !== incomingRefreshToken) {
      // OPTIONAL: لو حصل rotation قبل كده والقديم اتسرق →
      // بنلغي كل التوكنات عشان السارق ما يستفدش
      throw new AppError("Refresh token has been revoked", 401);
    }

    // ===============================================
    // الخطوة 4: التحقق من حالة الحساب
    // ===============================================
    if (user.isBanned) {
      throw new AppError("Account is banned", 403);
    }

    if (user.lockedUntil && user.lockedUntil > new Date()) {
      throw new AppError("Account is locked", 403);
    }

    // ===============================================
    // الخطوة 5: توليد توكنات جديدة (TOKEN ROTATION)
    // ===============================================
    // بنعمل token rotation: نولد access + refresh جدد
    const newAccessToken = tokenService.generateAccessToken({ id: user.id });
    const newRefreshToken = tokenService.generateRefreshToken({ id: user.id });

    // ===============================================
    // الخطوة 6: حفظ الـ refresh الجديد في DB (إلغاء القديم)
    // ===============================================
    await authModel.updateRefreshToken(user.id, newRefreshToken);

    // ===============================================
    // الخطوة 7: إرجاع النتيجة للـ Controller
    // ===============================================
    return {
      accessToken: newAccessToken,
      refreshToken: newRefreshToken,
    };
  }
}

export const authService = new AuthService();
🤔 الفرق الجوهري:
- Service بياخد incomingRefreshToken: string ← data من المستخدم
- Service بيرجع { accessToken, refreshToken } ← data عادية
- Service مش عنده أي فكرة عن Express (لا req, لا res, لا next)
- تقدر تختبر Service بكل سهولة: authService.refreshAccessToken("some-token")

4. auth.model.ts — طبقة قاعدة البيانات

// modules/auth/auth.model.ts
import prisma from "../../config/database";

class AuthModel {
  // ===============================================
  // كل دوال Prisma مركزة هنا
  // ===============================================

  async findById(id: string) {
    return prisma.user.findUnique({
      where: { id },
      select: {
        id: true,
        refreshToken: true,
        isBanned: true,
        lockedUntil: true,
      },
    });
  }

  async updateRefreshToken(userId: string, newRefreshToken: string) {
    return prisma.user.update({
      where: { id: userId },
      data: { refreshToken: newRefreshToken },
    });
  }

  async findUserByIdentifier(identifier: string) {
    return prisma.user.findFirst({
      where: {
        OR: [{ email: identifier }, { phone: identifier }],
      },
    });
  }
}

export const authModel = new AuthModel();
💡 ليه Data Layer منفصل؟
1. Testing: تقدر تستخدم Mock للـ Model بدون ما تحتاج قاعدة بيانات حقيقية
2. صيانة: لو غيرت ORM من Prisma لحاجة تانية (Drizzle, TypeORM) — التغيير في مكان واحد
3. وضوح: كل queries الـ DB مركزة — مش مبعثرة في Service
4. Performance: تقدر تضيف caching layer بسهولة (Redis) بين Service و Model

5. token.service.ts — توليد التوكنات

// modules/auth/token.service.ts
// أو ممكن يكون في shared/utils/tokens.ts لو أكتر من موديول محتاجه
import jwt from "jsonwebtoken";
import { ENV } from "../../config/env";

class TokenService {
  generateAccessToken(payload: object): string {
    return jwt.sign(payload, ENV.ACCESS_TOKEN_SECRET!, {
      expiresIn: "15m", // 15 دقيقة — مش 24 ساعة
    });
  }

  generateRefreshToken(payload: object): string {
    return jwt.sign(payload, ENV.REFRESH_TOKEN_SECRET!, {
      expiresIn: "7d", // 7 أيام
    });
  }

  verifyAccessToken(token: string) {
    return jwt.verify(token, ENV.ACCESS_TOKEN_SECRET!);
  }

  verifyRefreshToken(token: string) {
    return jwt.verify(token, ENV.REFRESH_TOKEN_SECRET!);
  }
}

export const tokenService = new TokenService();

// استخدم string durations مش milliseconds — أسهل وأوضوح:
// expiresIn: "15m"  (15 دقيقة) أفضل من expiresIn: 900000
// expiresIn: "7d"   (7 أيام)   أفضل من expiresIn: 604800000
⚠️ تعديل مهم: في المشروع الحالي، الـ Access Token صلاحيته 24 ساعة — ده كتير جدًا. الأحسن:
- Access Token: 15-30 دقيقة (يحد من الضرر لو اتسرق)
- Refresh Token: 7-30 يوم (حسب حاجة التطبيق)
الفكرة: لو اتسرق access token، الضرر محدود بربع ساعة. الـ refresh token يتجدد كل ما نستخدمه (rotation) فحتى لو اتسرق، أول مرة الضحية تعمل refresh، السارق بيقعد معاه توكن ملغي.

6. authenticate.middleware.ts — حماية الـ Routes

// shared/middleware/authenticate.middleware.ts
import { Request, Response, NextFunction } from "express";
import { AppError } from "../errors/AppError";
import { tokenService } from "../../modules/auth/token.service";

export const authenticate = (
  req: Request,
  _res: Response,
  next: NextFunction
) => {
  // ===============================================
  // 1. نجيب الـ access token
  // ===============================================
  const token =
    req.cookies?.accessToken ||
    (req.headers?.authorization?.startsWith("Bearer") &&
      req.headers.authorization.split(" ")[1]);

  if (!token) {
    throw new AppError("Authentication required", 401);
  }

  // ===============================================
  // 2. نتحقق من صحة الـ JWT
  // ===============================================
  try {
    const payload = tokenService.verifyAccessToken(token);
    // نحط بيانات المستخدم في req.user عشان الـ endpoint يستخدمها
    req.user = payload;
    next();
  } catch (error) {
    throw new AppError("Invalid or expired access token", 401);
  }
};
🔍 ازاي نستخدمها في الـ Routes:
import { authenticate } from "../../shared/middleware/authenticate.middleware";

// Routes محتاجة تسجيل دخول:
router.get("/profile", authenticate, profileController);
router.post("/logout", authenticate, logoutController);

// Routes عامة (زي refresh):
router.post("/refresh-accessToken", refreshAccessToken);
// مفيش authenticate هنا — مش محتاج access token

السببالشرحفي المشروع الحالي
Separation of Concerns
(فصل المسؤوليات)		 كل طبقة ليها مسؤولية واحدة واضحة. Controller ياخد request ويرد response. Service يعمل business logic. Model يتعامل مع DB. ❌ Class فيه Business Logic + Response Logic + DB queries كلها في نفس المكان
Testability
(قابلية الاختبار)		 تقدر تختبر Service بدون Express: authService.refreshAccessToken("fake-token") وتتأكد من النتيجة أو الخطأ. ❌ محتاج تعمل mock لـ req, res, next كاملين عشان تختبر دالة واحدة
Maintainability
(سهولة الصيانة)		 عايز تغير Prisma لـ Drizzle؟ عدل auth.model.ts فقط. عايز تغير JWT لـ别的 library؟ عدل token.service.ts فقط. ❌ Prisma queries مبعثرة جوا User.class.ts — لو عايز تغير ORM هتتعب
Developer Experience
(سهولة على المبرمج)		 أي مبرمج جديد يفتح المشروع يفهم: "خليني أشوف الـ Controller ← أشوف الـ Service ← خلاص فهمت". نمط مألوف. ❌ أنت شخصيًا اتلخبطت في الـ User.class وعملت سؤال — دليل إنه مش واضح
Scalability
(قابلية التوسع)		 لما عايز تفصل Auth Module لـ Microservice مستقبلًا، بتنسق فولدر modules/auth/ كله وتضيفله HTTP layer. ✅ الـ README بياخد ده في الاعتبار — لكن التطبيق الفعلي مش متوافق
